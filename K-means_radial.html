Assign entities left to k clusters by least distance criteria¶

In [7]:
def distance(a, b):
    '''
    returns (int / float):
            euqlidean distance between points
    parameters:
            a(positional, ndarray):
                first point
               
            b(positional, ndarray):
                second point
               
    One of points can be represented by tuple
    '''
    return np.sqrt(np.sum((a - b) ** 2, axis = 0))

def clustering(arr, P):
    #find distances from points to centroids, than find index of the minimum distance for every point
    return np.apply_along_axis(lambda x: np.apply_along_axis(lambda p: distance(x, p), 1, arr[P, :] ), 1, arr).argmin(1)

while True:
    
    prvC = cent.copy() #centroids on previous iteration

    for c in range(0, k): #for every cluster
        singleCluster = np.where(clusters == c) #entitys' indexes of currently procesed cluster

        cent[c, 0] = ar[singleCluster, 0].mean() # x of center of mass 
        cent[c, 1] = ar[singleCluster, 1].mean() # y of center of mass
        
       
    #if centroits has not changed after iteration
    if (cent == prvC).all():
        #plot dataset with separation clusters by color
